  <div class="container">

    <input id="itemInput" type="text" placeholder="Enter item ID (e.g., BOOSTER_COOKIE)" />

    <div id="results"></div>

  </div>


  <script>

    const input = document.getElementById("itemInput");

    const results = document.getElementById("results");


    input.addEventListener("change", async () => {

      const itemId = input.value.toUpperCase().trim();

      if (!itemId) return;


      results.innerHTML = "<p>Loading...</p>";


      try {

        const res = await fetch("https://api.hypixel.net/skyblock/bazaar");

        const data = await res.json();

        const item = data.products[itemId];


        if (!item) {

          results.innerHTML = "<p>Item not found.</p>";

          return;

        }


        const buyOrders = item.buy_summary;

        const sellOrders = item.sell_summary;


        const createTable = (orders, type) => {

          let totalCoins = 0;

          let totalAmount = 0;


          const rows = orders.map(o => {

            totalCoins += o.pricePerUnit * o.amount;

            totalAmount += o.amount;

            return `<tr><td>${o.pricePerUnit.toFixed(1)}</td><td>${o.amount}</td><td>${(o.pricePerUnit * o.amount).toLocaleString()}</td></tr>`;

          }).join("");


          return `

            <h2 class="section-title">${type}</h2>

            <table>

              <thead><tr><th>Price</th><th>Amount</th><th>Total</th></tr></thead>

              <tbody>${rows}</tbody>

              <tfoot><tr><th colspan="2">Total</th><th>${totalCoins.toLocaleString()}</th></tr></tfoot>

            </table>

          `;

        };


        const buyTable = createTable(buyOrders, "Buy Orders");

        const sellTable = createTable(sellOrders, "Sell Orders");


        results.innerHTML = buyTable + sellTable;

      } catch (err) {

        results.innerHTML = "<p>Error fetching data.</p>";

        console.error(err);

      }

    });

  </script>
