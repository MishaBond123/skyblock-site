<!DOCTYPE html>

<html>

<head>

  <title>SkyBlock Bazaar Tracker</title>

</head>

<body>

  <h1>SkyBlock Bazaar Prices</h1>

  <div id="prices">Loading...</div>


  <script>

    const items = ["TUNGSTEN_PLATE", "REFINED_UMBER", "GLACITE_AMALGAMATION"];

    fetch("https://api.hypixel.net/skyblock/bazaar")

      .then(res => res.json())

      .then(data => {

        let html = "<ul>";

        for (let item of items) {

          const q = data.products[item].quick_status;

          html += `<li><strong>${item.replaceAll("_", " ")}:</strong> Buy: ${q.buyPrice.toFixed(1)} | Sell: ${q.sellPrice.toFixed(1)}</li>`;

        }

        html += "</ul>";

        document.getElementById("prices").innerHTML = html;

      });

  </script>

</body>

</html>
