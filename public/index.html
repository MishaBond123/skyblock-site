  <div class="content">

    <h1>SkyBlock Bazaar Tracker</h1>

    <input type="text" id="itemInput" placeholder="Enter item ID (e.g., BOOSTER_COOKIE)">

    <div id="tables">Enter an item ID to see orders.</div>

  </div>


  <script>

    document.getElementById('itemInput').addEventListener('change', function () {

      const itemId = this.value.toUpperCase();

      fetch(`https://api.hypixel.net/skyblock/bazaar`)

        .then(res => res.json())

        .then(data => {

          const product = data.products[itemId];

          if (!product) {

            document.getElementById('tables').innerHTML = '<p>Invalid item ID or no data available.</p>';

            return;

          }


          const buyOrders = product.buy_summary;

          const sellOrders = product.sell_summary;


          const makeTable = (orders, type) => {

            let totalCoins = 0;

            let rows = orders.map(o => {

              const coins = o.pricePerUnit * o.amount;

              totalCoins += coins;

              return `<tr><td>${o.pricePerUnit.toFixed(1)}</td><td>${o.amount}</td><td>${coins.toLocaleString()}</td></tr>`;

            }).join('');


            return `

              <h2>${type} Orders</h2>

              <table>

                <thead><tr><th>Price</th><th>Amount</th><th>Total</th></tr></thead>

                <tbody>${rows}</tbody>

                <tfoot><tr><td colspan="2"><strong>Total</strong></td><td><strong>${totalCoins.toLocaleString()}</strong></td></tr></tfoot>

              </table>

            `;

          };


          const buyTable = makeTable(buyOrders, "Buy");

          const sellTable = makeTable(sellOrders, "Sell");


          document.getElementById('tables').innerHTML = buyTable + sellTable;

        });

    });

  </script>
